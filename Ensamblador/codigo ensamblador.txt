# Algoritmo para calcular el MCD de dos números en MIPS

# Registro $t0: primer número
# Registro $t1: segundo número
# Registro $t2: resto
# Registro $t3: temporal
# Registro $t4: MCD

# Inicializar los registros
addi $t2, $zero, 0   # Inicializar el registro del resto en 0
add $t4, $zero, $t0  # Inicializar el registro del MCD con el primer número

# Bucle para calcular el MCD
loop:
  beq $t1, $zero, end    # Si el segundo número es cero, terminar el bucle
  sw $t0, 0($sp)        # Guardar el primer número en la pila
  sw $t1, 4($sp)        # Guardar el segundo número en la pila
  div $t0, $t1          # Dividir el primer número por el segundo número
  mfhi $t2              # Guardar el resto en el registro $t2
  add $t3, $zero, $t0   # Guardar el primer número en el registro temporal
  lw $t0, 0($sp)        # Restaurar el primer número desde la pila
  lw $t1, 4($sp)        # Restaurar el segundo número desde la pila
  move $t0, $t1         # Mover el segundo número al registro del primer número
  move $t1, $t2         # Mover el resto al registro del segundo número
  j loop               # Volver al inicio del bucle

# Fin del bucle, el MCD se encuentra en el registro $t4
end:
  nop                   # No operation
